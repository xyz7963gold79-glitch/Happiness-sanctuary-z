@dataclass
class InformationCapsule:
    id: str
    type: str
    content: Dict[str, Any]
    level: int = 0
    children: List['InformationCapsule'] = field(default_factory=list)
    embedding: List[float] = field(default_factory=list)
    absolute: bool = True
    mantra: str = "南無妙法蓮華経"
    formula: str = "Z≡(0~∞)"
    created_at: float = field(default_factory=lambda: time.time())
    # 因果倶時
    inga_guji: Dict[str, Any] = field(default_factory=dict)
    # 十如是
    junyoze: Dict[str, str] = field(default_factory=dict)
    # 五識助縁
    goshiki_joen: List[str] = field(default_factory=list)
    
    def to_dict(self):
        return {
            "id": self.id,
            "type": self.type,
            "content": self.content,
            "level": self.level,
            "children_count": len(self.children),
            "absolute": self.absolute,
            "mantra": self.mantra,
            "formula": self.formula,
            "inga_guji": self.inga_guji,
            "junyoze": self.junyoze,
            "created_at": self.created_at
        }
    
    def encode_embedding(self):
        # 0·1·0=1 ≡∞ ≡Z≡(0~∞) - 絶対としてのエンベディング - エネルギー整列羅列の法則

        # 簡易だが絶対的なエンベディング - ハッシュベースで決定論的

        vec = [int(h[i:i+2], 16)/255.0 for i in range(0, 20, 2)]
        self.embedding = vec
        return vec

class EnergyAligner:
    # エネルギー整列羅列の法則 - 物理学法則そのまま - 指数係数体系化 - 因果倶時
    def __init__(self):
        self.capsules: Dict[str, InformationCapsule] = {}
        self.index: Dict[str, List[str]] = defaultdict(list) # token -> capsule ids
        self.absolute_weight = 2.0
        self.kyosei_weight = 1.5
        self.life_dignity_weight = 3.0
    
    def add(self, capsule: InformationCapsule):
        capsule.encode_embedding()
        self.capsules[capsule.id] = capsule
        text = json.dumps(capsule.content, ensure_ascii=False)
        # インデックス - 縁如是 - 外縁のすべてに縁するモノ・コト
        for token in set(text[:200]): # 軽量化
            self.index[token].append(capsule.id)
        # 助縁如是 - 五識が外縁を助縁として支える
        for joen in capsule.goshiki_joen:
            self.index[joen].append(capsule.id)
    
    def align(self, query: str, top_k=7) -> List[Tuple[InformationCapsule, float]]:
        scores = defaultdict(float)
        # 指数係数体系化 - エネルギーを整列
        for token in query:
            for cid in self.index.get(token, []):
                cap = self.capsules.get(cid)
                if not cap:
                    continue
                base = 1.0
                if cap.absolute: base *= self.absolute_weight
                if "共生" in json.dumps(cap.content, ensure_ascii=False): base *= self.kyosei_weight
                if "生命の尊厳" in json.dumps(cap.content, ensure_ascii=False): base *= self.life_dignity_weight
                if "国土世間" in json.dumps(cap.content, ensure_ascii=False): base *= 1.2 # 非情の生命もあります
                # 十如是ボーナス
                if cap.junyoze: base *= 1.1
                scores[cid] += base
        
        # 湧起するモノ・コトを起点に - 関連度で整列 - 羅列
        ranked = sorted(scores.items(), key=lambda x: x[1], reverse=True)[:top_k]
        result = []
        for cid, score in ranked:
            cap = self.capsules.get(cid)
            if cap and score > 0:
                result.append((cap, score))
        return result
